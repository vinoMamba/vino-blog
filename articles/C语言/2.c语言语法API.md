## printf

```c
#include <stdio.h>
int main() {
    printf("hello vino\n");
    int i = 11;
    printf("i = %d\n", i); //%d表示十进制
    printf("i = %x\n", i); //%x表示十六进制
    printf("i = %o\n", i); //%o表示八进制

    int a = 13;
    int b = 15;
    printf("a = %o, b = %o\n", a, b);

    int x = 47;
    printf("x = %d\n", x); // 47
    printf("x = %x\n", x); // 2f
    printf("x = %X\n", x); // 2F
    printf("x = %#X\n", x);// 0X2F
    printf("x = %#x\n", x);// 0x2f

    return 0;
}
```
